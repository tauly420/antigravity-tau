```typescript
const handleExportPdf = async () => {
    setExportStatus('exporting');
    setExportError(null);
    try {
        // Normalize analysis data for the backend
        const rawData = autolabResults ? (autolabResults as Record<string, unknown>) : {};
        const analysisData = normalizeAnalysisData(rawData);

        const blob = await exportReportPdf({
            sections: editableSections,
            title_page: {
                studentName: titlePageData.studentName,
                studentId: titlePageData.studentId,
                labPartner: titlePageData.labPartner,
                courseName: titlePageData.courseName,
                experimentTitle: titlePageData.experimentTitle,
                date: titlePageData.date,
            },
            plots: {
                fit: plotImages.fit,
                residuals: plotImages.residuals,
            },
            template: selectedTemplate,
            language: language,
            analysis_data: analysisData,
        });

        // Trigger browser download
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = 'lab-report.pdf';
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);

        setExportStatus('idle');
    } catch (err: unknown) {
        const msg = err instanceof Error ? err.message : 'PDF generation failed. Check that all required fields are filled and try again.';
        setExportError(msg);
        setExportStatus('error');
    }
};
```

**4. Add plot capture logic in ReportBeta.tsx:**

Import Plotly:
```typescript
import Plotly from 'plotly.js-dist-min';
```

Add a `useEffect` that captures plots when `autolabResults` changes and the component mounts. The plots exist as DOM elements on the AutoLab page. Since `autolabResults` carries data but not DOM refs, we need to capture images from the plotly chart divs IF they exist on the page:

```typescript
const { autolabResults, plotImages, setPlotImages } = useAnalysis();

// Attempt to capture plot images from AutoLab page DOM on mount
useEffect(() => {
    const capturePlots = async () => {
        try {
            // Look for Plotly chart divs (they have class "js-plotly-plot")
            const plotDivs = document.querySelectorAll('.js-plotly-plot');
            if (plotDivs.length >= 1) {
                const fitImg = await Plotly.toImage(plotDivs[0] as any, { format: 'png', width: 700, height: 400, scale: 2 });
                const residualsImg = plotDivs.length >= 2
                    ? await Plotly.toImage(plotDivs[1] as any, { format: 'png', width: 700, height: 300, scale: 2 })
                    : null;
                setPlotImages({ fit: fitImg, residuals: residualsImg });
            }
        } catch (e) {
            console.warn('Could not capture plot images:', e);
        }
    };

    // Only attempt if we have autolab results and no images captured yet
    if (autolabResults && !plotImages.fit) {
        capturePlots();
    }
}, [autolabResults, plotImages.fit, setPlotImages]);
```

Note: This capture works when user navigates from AutoLab page (where plot DOM exists) to Report page if the plots are still in DOM. If they are not (SPA routing unmounted them), plotImages will be null and PlotThumbnail will show the "No plot available" placeholder. This is the expected graceful degradation per D-07 and research open question 3.
