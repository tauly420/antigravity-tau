    **Step 1: Create frontend/src/types/report.ts** (per D-06, D-07, D-08, D-09, D-10, D-11)

    Define these exact exported interfaces:

    ```typescript
    export interface FitParameter {
      name: string;
      value: number;
      uncertainty: number;
      rounded: string;    // "49.8 +/- 0.5" from scientific_round
      latex: string;       // "$k = 49.8 \pm 0.5$" for KaTeX
    }

    export interface GoodnessOfFit {
      chiSquaredReduced: number | null;
      rSquared: number | null;
      pValue: number | null;
      dof: number | null;
    }

    export interface FitSection {
      modelName: string;
      parameters: FitParameter[];
      goodnessOfFit: GoodnessOfFit;
      xData: number[];
      yData: number[];
      xErrors: number[] | null;
      yErrors: number[] | null;
      xFit: number[];
      yFit: number[];
      residuals: number[];
    }

    export interface FormulaSection {
      expression: string;
      value: number;
      uncertainty: number;
      formatted: string;  // "9.81 +/- 0.10"
      latex: string;       // "$g = 9.81 \pm 0.10$"
    }

    export interface NSigmaSection {
      nSigma: number;
      verdict: string;
      theoreticalValue: number;
      theoreticalUncertainty: number;
    }

    export interface ReportAnalysisData {
      fit: FitSection | null;
      formula: FormulaSection | null;
      nsigma: NSigmaSection | null;
      summary: string | null;
      instructions: string;
      filename: string;
    }
    ```

    **Step 3: Create backend/utils/report_normalizer.py** (per D-07, D-10)

    Function signature: `def normalize_autolab_result(state: dict, steps: list, instructions: str, filename: str) -> dict`

    Implementation details:
    - Import `scientific_round` from `utils.calculations`
    - Import `math` for `math.isfinite` checks
    - For fit section: iterate `parameter_names` with zip over `parameters` and `uncertainties`. For each, call `scientific_round(val, unc)` to get the rounded string. Build latex as `f"${name} = {rounded_str}$"` (replacing +/- with \\pm).
    - For goodnessOfFit: read `reduced_chi_squared`, `r_squared`, `p_value`, `dof` from fit dict. Convert non-finite values to None using: `v if (v is not None and math.isfinite(v)) else None`.
    - For formula: read `value`, `uncertainty`, `formatted`, `expression`. Build latex as `f"${expression} = {formatted}$"` with \\pm substitution.
    - For nsigma: map `n_sigma` -> `nSigma`, `theoretical_value` -> `theoreticalValue`, `theoretical_uncertainty` -> `theoreticalUncertainty`.
    - For summary: iterate steps in reverse, find first with `step=="summary"` and non-empty `message`.
    - All output keys MUST be camelCase. No snake_case keys at any level.
    - Sections with errors (dict containing "error" key) should be treated as None.
